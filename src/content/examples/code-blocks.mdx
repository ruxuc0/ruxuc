## javascript examples

```javascript
function greet(name) {
}

const user = "world";
greet(user);

// arrow functions
const multiply = (a, b) => a * b;

// async/await
async function fetchData() {

```

## python examples

```python
def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# list comprehension
squares = [x**2 for x in range(10)]

# dictionary
config = {
    "host": "localhost",
    "port": 3000,
    "debug": True
}

class Calculator:
    def __init__(self):
        self.result = 0
    
    def add(self, value):
        self.result += value
        return self
```

## bash/shell examples

```bash

# variables
name="astro"
version=$(npm list astro --depth=0 | grep astro)

# functions
deploy() {
  echo "deploying to production..."
  npm run build
  rsync -av dist/ user@server:/var/www/
}

# conditionals
if [ $# -eq 0 ]; then
  echo "no arguments provided"
  exit 1
fi

# loops
for file in *.md; do
  echo "processing $file"
  pandoc "$file" -o "output/${file%.md}.html"
done
```
